void removeSessionCookie() {
  try {
    response.cookies.add(
      Cookie(sessionLoginKey, 'false')
        ..httpOnly = false
        ..path = '/'
        ..expires = DateTime.now(),
    );
    response.cookies.add(
      Cookie(sessionKey, '-')
        ..path = '/'
        ..expires = DateTime.now(),
    );
  } catch (e) {
  }
}
